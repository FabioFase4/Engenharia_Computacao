💻 O Código do Engenheiro

Simulação de manipulação de bits em um registrador de hardware.

hardware_sim.cpp 
#include <iostream>
// Registrador volátil (Hardware real)
volatile int* PORTA_A = new int(0);

int main() {
    // Ligar Pino 3 (Mascara 00001000)
    *PORTA_A |= (1 << 3); 

    // Ligar Pinos 0 e 5
    *PORTA_A |= (1 << 0) | (1 << 5);

    // Desligar Pino 3 via AND inverso
    *PORTA_A &= ~(1 << 3);

    return 0;
}
